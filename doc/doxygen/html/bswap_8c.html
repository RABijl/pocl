#define GENERIC_BYTESWAP (   __DTYPE,
  __WORD 
)
Value:
do { \
unsigned __i; \
union _word_union \
{ \
__DTYPE full_word; \
unsigned char bytes[sizeof(__DTYPE)]; \
} __old, __neww; \
\
__old.full_word = __WORD; \
for (__i = 0; __i < sizeof(__DTYPE); ++__i) \
__neww.bytes[__i] = __old.bytes[sizeof(__DTYPE) - __i - 1]; \
__WORD = __neww.full_word; \
} while(0)

inplace endian byte swap macro

Parameters
__DTYPE[in] word datatype
__WORD[in/out] data to be swapped in place
